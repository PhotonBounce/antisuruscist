SECTION 3: GAME OVERVIEW
────────────────────────────────────────────────────────────────────────────────

Side-scrolling zombie wave shooter.
- Zombies walk right→left, player clicks/taps to shoot
- 4 waves per game; between waves an inventory/armory modal opens
- Player earns credits (💰) from kills, arcoins (🪙 ARC) per wave cleared
- Credits buy weapon upgrades and ammo refills
- ARC is the play-to-earn token (future Polygon ERC-20)
- God mode: unlocks all weapons, all upgrades, infinite ammo

Canvas dimensions: 1024×550 (scaled to fit viewport via CSS --canvas-scale)

  Function to implement:
    function generateARCTokenCanvas(tokenId, waveNum) {
      const c = document.createElement('canvas');
      c.width = c.height = 64;
      const ctx = c.getContext('2d');
      // 1. Hexagon path (6 sides)
      const cx = 32, cy = 32, r = 28;
      ctx.beginPath();
      for (let i = 0; i < 6; i++) {
        const a = (Math.PI / 3) * i - Math.PI / 6;
        i === 0 ? ctx.moveTo(cx + r*Math.cos(a), cy + r*Math.sin(a))
                : ctx.lineTo(cx + r*Math.cos(a), cy + r*Math.sin(a));
      }
      ctx.closePath();
      // 2. Radial gradient — hue unique per tokenId
      const hue = (tokenId * 37 + waveNum * 13) % 60 + 30; // 30-90 (gold range)
      const grad = ctx.createRadialGradient(cx-8, cy-8, 4, cx, cy, r);
      grad.addColorStop(0, `hsl(${hue}, 100%, 80%)`);
      grad.addColorStop(0.6, `hsl(${hue}, 90%, 55%)`);
      grad.addColorStop(1, `hsl(${hue-15}, 80%, 30%)`);
      ctx.fillStyle = grad;
      ctx.fill();
      // 3. Gold border stroke
      ctx.strokeStyle = `hsl(${hue+10}, 100%, 70%)`;
      ctx.lineWidth = 2;
      ctx.stroke();
      // 4. Inner emboss ring
      ctx.beginPath();
      for (let i = 0; i < 6; i++) {
        const a = (Math.PI / 3) * i - Math.PI / 6;
        const ri = r * 0.75;
        i === 0 ? ctx.moveTo(cx + ri*Math.cos(a), cy + ri*Math.sin(a))
                : ctx.lineTo(cx + ri*Math.cos(a), cy + ri*Math.sin(a));
      }
      ctx.closePath();
      ctx.strokeStyle = `hsla(${hue+20}, 100%, 85%, 0.5)`;
      ctx.lineWidth = 1;
      ctx.stroke();
      // 5. "ARC" text
      ctx.fillStyle = `hsl(${hue+15}, 100%, 90%)`;
      ctx.font = 'bold 11px Oswald, Impact, sans-serif';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.shadowColor = `hsl(${hue}, 100%, 30%)`;
      ctx.shadowBlur = 3;
      ctx.fillText('ARC', cx, cy - 4);
      // 6. Wave/token number
      ctx.font = '7px Oswald, sans-serif';
      ctx.shadowBlur = 0;
      ctx.fillStyle = `hsla(${hue+10}, 80%, 85%, 0.8)`;
      ctx.fillText('W' + waveNum, cx, cy + 6);
      return c.toDataURL('image/png');
    }
